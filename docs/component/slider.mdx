# Slider

    const [value, setValue] = useState(30)
    return (
      <>
        <Slider defaultValue={30} />
        <br />
        <Slider
          value={value}
          onChange={(value) => {
            console.log(value)
            setValue(value)

      </>
    )
  }}
</Playground>
