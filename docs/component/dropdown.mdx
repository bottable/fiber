### Input Dropdown

<Playground>
  {() => {
    const [value, setValue] = useState('')
    const menuArray = ['1st menu item', '2nd menu item', '3rd menu item']
    const menu = (
      <Menu>
        {menuArray.map((menuItem, idx) => (
          <Menu.Item
            onClick={() => {
              setValue(menuItem)
            }}
            key={idx}
          >
            {menuItem}
          </Menu.Item>
        ))}
      </Menu>
    )
    return (
      <Dropdown.Input
        overlay={menu}
        value={value}
        onChange={(event) => setValue(event.target.value)}
      />
    )
  }}
</Playground>
