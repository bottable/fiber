## Colors

<Playground>
  {() => {
    const colors = [
      'blue',
      'green',
      'magenta',
      'neutral',
      'orange',
      'purple',
      'red',
      'teal',
      'yellow'
    ]
    const customColors = ['#f50', '#2db7f5', '#87d068', '#108ee9']
    return (
      <div>
        <Divider orientation='left'>Presets</Divider>

            </Tooltip>
          ))}
        </Space>
        <Divider orientation='left'>Custom</Divider>
        <Space>

            </Tooltip>
          ))}
        </Space>
      </div>
    )
  }}
</Playground>

## Example

<Playground>
  {() => {
    const [dropdownValue, setDropdownValue] = useState('has changed')
    const menuArray = ['has changed', 'has increased', 'has decreased']
    const menu = (
      <Menu>
        {menuArray.map((menuItem, idx) => (
          <Menu.Item
            onClick={() => {
              setDropdownValue(menuItem)
            }}
            key={idx}
          >
            {menuItem}
          </Menu.Item>
        ))}
      </Menu>
    )
    return (
      <Collapse style={{ width: 270 }}>
        <Collapse.Panel
          header={
            <Input
              defaultValue='Category'
              camouflage
              style={{ width: '100%', fontWeight: 700 }}
            />
          }
          extra={<DeleteIcon />}
        >
          <Space
            direction='vertical'
            style={{ width: '100%', color: '#828282' }}
          >
            <Text style={{ width: '100%' }}>
              Notify me when selected value{' '}
              <Tooltip
                title='Lorem ipsum dolor sit amet nsectetur adipisc lorem ipsu...'
                placement='bottom'
              >
                <InfoIcon
                  style={{
                    color: 'black',
                    width: 12,
                    height: 12,
                    verticalAlign: 'top'
                  }}
                />
              </Tooltip>
            </Text>
            <Dropdown.Button
              overlay={menu}
              style={{ width: '100%' }}
              button={{
                endIcon: <ExpandMoreIcon />,
                block: true
              }}
            >
              {dropdownValue}
            </Dropdown.Button>
          </Space>
        </Collapse.Panel>
      </Collapse>
    )
  }}
</Playground>
